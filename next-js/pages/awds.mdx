# Colors

export const Colors = () => {
  const { theme } = useThemeUI();
  const actualColors = Object.keys(theme.colors).filter((c) => c !== 'modes');
  return (
    <Grid sx={{ marginTop: 3 }} columns={[1, 4]} gap={[1, 3]}>
      {actualColors.map((color) => (
        <Grid key={color} columns={[2, 1]}>
          <Box
            sx={{
              borderColor: 'highlight',
              borderWidth: '1px',
              borderStyle: 'solid',
              backgroundColor: theme.colors[color],
              height: ['space.4', 'space.6'],
              width: '100%',
            }}
          />
          <Box>
            <Text>{color}</Text>
            <Text variant="label">{theme.colors[color]}</Text>
          </Box>
        </Grid>
      ))}
    </Grid>
  );
};

<Colors />
